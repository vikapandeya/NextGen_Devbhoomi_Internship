Async / Fetch
// Fetch API
const getData = async () => {
  try {
    const res = await fetch('/api/tasks');
    if (!res.ok) throw new Error('Error');
    const data = await res.json();
    return data;
  } catch (err) {
    console.error(err);
  }
};

// POST request
await fetch('/api/tasks', {
  method: 'POST',
  headers: {'Content-Type': 'application/json'},
  body: JSON.stringify({ title: 'New task' }),
});
React — Hooks & Patterns
Component Template
import { useState, useEffect } from 'react';

function MyComponent({ title, onAction }) {
  const [data, setData] = useState(null);

  useEffect(() => {
    // Runs once on mount
    fetchData().then(setData);
  }, []);

  if (!data) return <p>Loading...</p>;

  return (
    <div>
      <h2>{title}</h2>
      {data.map(item => (
        <div key={item.id}>{item.name}</div>
      ))}
    </div>
  );
}

export default MyComponent;
useState Patterns
// Basic
const [count, setCount] = useState(0);
setCount(count + 1);

// Object state
const [form, setForm] = useState({ name:'', email:'' });
setForm({ ...form, name: 'Priya' });

// Array state
const [items, setItems] = useState([]);
// Add item:
setItems([...items, newItem]);
// Remove item:
setItems(items.filter(i => i.id !== id));
// Update item:
setItems(items.map(i =>
  i.id === id ? { ...i, done: true } : i
));
Conditional Rendering

// Ternary
{loading ? <Spinner /> : <Content />}

// Early return
if (error) return <ErrorMessage msg={error} />;
if (loading) return <Spinner />;

// Multiple conditions

{status === 'loading' && <Spinner />}
{status === 'success' && <SuccessView />}
Node.js / Express — Quick Reference
Server Setup
const express = require('express');
const cors    = require('cors');
require('dotenv').config();

const app = express();

app.use(cors());           // Allow CORS
app.use(express.json());   // Parse JSON body

app.listen(3001, () =>
  console.log('Server on :3001'));
Route Patterns
// GET all
app.get('/api/items', async (req, res) => {
  const [rows] = await db.query('SELECT * FROM items');
  res.json(rows);
});

// GET one by id
app.get('/api/items/:id', async (req, res) => {
  const { id } = req.params;
  const [rows] = await db.query(
    'SELECT * FROM items WHERE id = ?', [id]);
  if (!rows.length) return res.status(404).json({error:'Not found'});
  res.json(rows[0]);
});

Git — Essential Commands
Daily Workflow

git add filename.js         # stage one file
git add .                   # stage all changes
git commit -m "Add login"   # commit with message
git push                    # push to remote
git pull                    # pull latest changes
Branching
git branch feature/login   # create branch
git checkout feature/login # switch to branch
git checkout -b fix/bug    # create + switch

git merge feature/login    # merge into current
git branch -d feature/login # delete branch

git log --oneline          # see commit history
