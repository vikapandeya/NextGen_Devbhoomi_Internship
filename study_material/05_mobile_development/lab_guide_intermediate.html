News Reader — fetch, Loading States & Pull-to-Refresh

⏱ 75 min
Goal: Fetch live headlines from NewsAPI, show loading/error/data states, and implement pull-to-refresh. Tap an article to open it in a browser.
Setup: Get a free API key from newsapi.org. Add it as a constant — in production you would store this in an environment variable, never commit API keys to Git.

Step 1 — Install dependencies

npm install axios
npx expo install expo-web-browser

Step 2 — NewsScreen.js

import { useState, useEffect, useCallback } from 'react';
import { View, Text, FlatList, Image, TouchableOpacity, StyleSheet, ActivityIndicator } from 'react-native';
import axios from 'axios';
import * as WebBrowser from 'expo-web-browser';

const API_KEY = 'YOUR_NEWSAPI_KEY_HERE'; // replace with your key
const API_URL = `https://newsapi.org/v2/top-headlines?country=in&apiKey=${API_KEY}`;

export default function NewsScreen() {
  const [articles, setArticles] = useState([]);
  const [isLoading, setIsLoading] = useState(true);
  const [error, setError] = useState(null);
  const [refreshing, setRefreshing] = useState(false);

  const fetchNews = useCallback(async (isRefresh = false) => {
    if (isRefresh) setRefreshing(true);
    try {
      const { data } = await axios.get(API_URL);
      setArticles(data.articles.filter(a => a.title && a.urlToImage));
      setError(null);
    } catch {
      setError('Could not load news. Check your connection and tap to retry.');
    } finally {
      setIsLoading(false); setRefreshing(false);
    }
  }, []);

  useEffect(() => { fetchNews(); }, [fetchNews]);

  if (isLoading) return (
    <View style={styles.center}>
      <ActivityIndicator size="large" color="#1B3A6B" />
      <Text style={styles.loadingText}>Fetching headlines...</Text>
    </View>
  );

  if (error) return (
    <View style={styles.center}>
      <Text style={styles.errorText}>{error}</Text>
      <TouchableOpacity style={styles.retryBtn} onPress={() => { setIsLoading(true); fetchNews(); }}>
        <Text style={styles.retryText}>Retry</Text>
      </TouchableOpacity>
    </View>
  );

  return (
    <FlatList
      data={articles}
      keyExtractor={a => a.url}
      refreshing={refreshing}
      onRefresh={() => fetchNews(true)}
      contentContainerStyle={{ padding: 16 }}
      renderItem={({ item }) => (
        <TouchableOpacity style={styles.card} onPress={() => WebBrowser.openBrowserAsync(item.url)} activeOpacity={0.85}>
          {item.urlToImage && <Image source={{ uri: item.urlToImage }} style={styles.image} />}
          <View style={styles.cardBody}>
            <Text style={styles.source}>{item.source.name}</Text>
            <Text style={styles.title} numberOfLines={2}>{item.title}</Text>
            <Text style={styles.date}>{new Date(item.publishedAt).toLocaleDateString('en-IN')}</Text>
          </View>
        </TouchableOpacity>
      )}
    />
  );
}

const styles = StyleSheet.create({
  center:      { flex: 1, justifyContent: 'center', alignItems: 'center', padding: 32 },
  loadingText: { marginTop: 12, fontSize: 14, color: '#5A6070' },
  errorText:   { fontSize: 14, color: '#C62828', textAlign: 'center', marginBottom: 16 },
  retryBtn:    { backgroundColor: '#1B3A6B', paddingHorizontal: 28, paddingVertical: 12, borderRadius: 8 },
  retryText:   { color: 'white', fontWeight: '700' },
  card:        { backgroundColor: 'white', borderRadius: 14, marginBottom: 14, overflow: 'hidden',
                 shadowColor: '#1B3A6B', shadowOffset: {width:0,height:2}, shadowOpacity: 0.08, shadowRadius: 8, elevation: 3 },
  image:       { width: '100%', height: 180 },
  cardBody:    { padding: 14 },
  source:      { fontSize: 11, fontWeight: '700', color: '#C8941A', textTransform: 'uppercase', marginBottom: 4 },
  title:       { fontSize: 15, fontWeight: '700', color: '#1B3A6B', lineHeight: 21, marginBottom: 8 },
  date:        { fontSize: 11, color: '#9AA0AE' },

  Clouds:       { bg: '#ECEFF1', icon: '☁️' },

  const [weather, setWeather]   = useState(null);
  const [isLoading, setIsLoading] = useState(false);
  const [error, setError]       = useState(null);

      setWeather(data);

      setError('Could not fetch weather. Check your API key and connection.');

            <Text style={styles.btnText}>📍 Get My Weather</Text>

        <View style={styles.center}>
          <ActivityIndicator size="large" color="#1B3A6B" />

          <Text style={styles.icon}>{theme.icon}</Text>

          <Text style={styles.temp}>{Math.round(weather.main.temp)}°C</Text>
          <Text style={styles.condition}>{weather.weather[0].description}</Text>

Goal: Build a photo grid app — pick images from the gallery or take new photos with the camera. Show them in a 2-column FlatList grid with delete support.

Step 2 — PhotoGridScreen.js

import { useState } from 'react';
import {
  View, Text, Image, FlatList, TouchableOpacity,
  StyleSheet, Alert, Dimensions
} from 'react-native';
import * as ImagePicker from 'expo-image-picker';

const { width } = Dimensions.get('window');
const PHOTO_SIZE = (width - 48) / 2; // 2 columns with gaps

export default function PhotoGridScreen() {
  const [photos, setPhotos] = useState([]);

  async function pickFromGallery() {
    const { status } = await ImagePicker.requestMediaLibraryPermissionsAsync();
    if (status !== 'granted') { Alert.alert('Permission needed', 'Allow photo library access in Settings.'); return; }
    const result = await ImagePicker.launchImageLibraryAsync({
      mediaTypes: ImagePicker.MediaTypeOptions.Images,
      allowsMultipleSelection: true,
      quality: 0.7,
    });
    if (!result.canceled) {
      const newPhotos = result.assets.map(a => ({ id: a.uri, uri: a.uri }));
      setPhotos(prev => [...newPhotos, ...prev]);
    }
  }

  async function takePhoto() {
    const { status } = await ImagePicker.requestCameraPermissionsAsync();
    if (status !== 'granted') { Alert.alert('Camera access needed', 'Allow camera access in Settings.'); return; }
    const result = await ImagePicker.launchCameraAsync({
      allowsEditing: true, aspect: [1, 1], quality: 0.8,
    });
    if (!result.canceled) {
      setPhotos(prev => [{ id: result.assets[0].uri, uri: result.assets[0].uri }, ...prev]);
    }
  }

  function deletePhoto(id) {
    Alert.alert('Remove photo?', '', [
      { text: 'Cancel' },
      { text: 'Remove', style: 'destructive', onPress: () => setPhotos(p => p.filter(ph => ph.id !== id)) },
    ]);
  }

  return (
    <View style={styles.screen}>
      <View style={styles.toolbar}>
        <TouchableOpacity style={styles.toolBtn} onPress={pickFromGallery}>
          <Text style={styles.toolText}>🖼️ Gallery</Text>
        </TouchableOpacity>
        <TouchableOpacity style={[styles.toolBtn, styles.toolBtnPrimary]} onPress={takePhoto}>
          <Text style={[styles.toolText, styles.toolTextPrimary]}>📷 Camera</Text>
        </TouchableOpacity>
      </View>

      {photos.length === 0 ? (
        <View style={styles.empty}>
          <Text style={styles.emptyText}>No photos yet.{'\n'}Tap Gallery or Camera above.</Text>
        </View>
      ) : (
        <FlatList
          data={photos}
          keyExtractor={p => p.id}
          numColumns={2}
          columnWrapperStyle={styles.row}
          contentContainerStyle={styles.grid}
          renderItem={({ item }) => (
            <TouchableOpacity onLongPress={() => deletePhoto(item.id)} activeOpacity={0.9}>
              <Image source={{ uri: item.uri }} style={styles.photo} />
            </TouchableOpacity>
          )}
        />
      )}
    </View>
  );
}

const styles = StyleSheet.create({
  screen:          { flex: 1, backgroundColor: '#F8F6F0' },
  toolbar:         { flexDirection: 'row', gap: 12, padding: 16, backgroundColor: 'white',
                     borderBottomWidth: 1, borderBottomColor: '#EAE7E0' },
  toolBtn:         { flex: 1, borderWidth: 1.5, borderColor: '#1B3A6B', borderRadius: 10,
                     paddingVertical: 10, alignItems: 'center' },
  toolBtnPrimary:  { backgroundColor: '#1B3A6B', borderColor: '#1B3A6B' },
  toolText:        { fontSize: 14, fontWeight: '700', color: '#1B3A6B' },
  toolTextPrimary: { color: 'white' },
  empty:           { flex: 1, justifyContent: 'center', alignItems: 'center', padding: 40 },
  emptyText:       { fontSize: 15, color: '#9AA0AE', textAlign: 'center', lineHeight: 24 },
  grid:            { padding: 16 },
  row:             { gap: 16, marginBottom: 16 },
  photo:           { width: PHOTO_SIZE, height: PHOTO_SIZE, borderRadius: 12 },
});
Checkpoint: Gallery picker opens, multiple photos can be selected, Camera opens and adds the photo to the grid, long-pressing a photo offers to delete it.

SQLite Task Manager — Real Local Database

⏱ 90 min
Goal: Build a task manager backed by an SQLite database. Supports create, toggle done, filter (All/Active/Completed), and delete. Data survives app restarts.

Step 1 — Install

npx expo install expo-sqlite

Step 2 — useTasksDb.js (custom hook)

import { useState, useEffect } from 'react';
import * as SQLite from 'expo-sqlite';

let db;

export function useTasksDb() {
  const [tasks, setTasks] = useState([]);
  const [filter, setFilter] = useState('all'); // 'all' | 'active' | 'done'

  useEffect(() => {
    (async () => {
      db = await SQLite.openDatabaseAsync('tasks.db');
      await db.execAsync(
        'CREATE TABLE IF NOT EXISTS tasks (id INTEGER PRIMARY KEY AUTOINCREMENT, title TEXT NOT NULL, done INTEGER DEFAULT 0);'
      );
      await loadTasks();
    })();
  }, []);

  async function loadTasks() {
    const rows = await db.getAllAsync('SELECT * FROM tasks ORDER BY id DESC;');
    setTasks(rows);
  }

  async function addTask(title) {
    if (!title.trim()) return;
    await db.runAsync('INSERT INTO tasks (title) VALUES (?);', [title.trim()]);
    await loadTasks();
  }

  async function toggleTask(id, currentDone) {
    await db.runAsync('UPDATE tasks SET done=? WHERE id=?;', [currentDone ? 0 : 1, id]);
    await loadTasks();
  }

  async function deleteTask(id) {
    await db.runAsync('DELETE FROM tasks WHERE id=?;', [id]);
    await loadTasks();
  }

  const filteredTasks = tasks.filter(t => {
    if (filter === 'active') return !t.done;
    if (filter === 'done')   return t.done;
    return true;
  });

  return { tasks: filteredTasks, filter, setFilter, addTask, toggleTask, deleteTask };
}

import { useState, useEffect } from 'react';
import { View, Text, TextInput, FlatList, TouchableOpacity, StyleSheet } from 'react-native';
import { useQuery, useMutation, useQueryClient } from '@tanstack/react-query';
import NetInfo from '@react-native-community/netinfo';

// Mock API — replace with a real backend URL in production
const MOCK_ITEMS = [
  { id: 1, name: 'Basmati Rice', qty: '2 kg' },
  { id: 2, name: 'Ghee', qty: '500 g' },
  { id: 3, name: 'Honey (Uttarakhand wild)', qty: '1 bottle' },
];
const fetchItems  = () => Promise.resolve(MOCK_ITEMS);
const createItem  = (item) => Promise.resolve({ ...item, id: Date.now() });

export default function ShoppingListScreen() {
  const [input, setInput] = useState('');
  const [isOnline, setIsOnline] = useState(true);
  const qc = useQueryClient();

  useEffect(() => {
    return NetInfo.addEventListener(s => setIsOnline(!!s.isConnected));
  }, []);

  const { data: items = [], isLoading } = useQuery({
    queryKey: ['shopping'],
    queryFn: fetchItems,
    staleTime: 1000 * 60 * 2,
  });

  const addMutation = useMutation({
    mutationFn: (name) => createItem({ name, qty: '1' }),
    onMutate: async (name) => {
      // optimistic update
      await qc.cancelQueries({ queryKey: ['shopping'] });
      const prev = qc.getQueryData(['shopping']);
      qc.setQueryData(['shopping'], old => [...(old || []), { id: Date.now(), name, qty: '1' }]);
      return { prev };
    },
    onError: (_, __, ctx) => qc.setQueryData(['shopping'], ctx.prev),
    onSettled: () => qc.invalidateQueries({ queryKey: ['shopping'] }),
  });

  return (
    <View style={styles.screen}>
      {!isOnline && (
        <View style={styles.offlineBanner}>
          <Text style={styles.offlineText}>📵 Offline Mode — showing cached data</Text>
        </View>
      )}
      <View style={styles.inputRow}>
        <TextInput style={styles.input} value={input} onChangeText={setInput}
          placeholder="Add item to list..." returnKeyType="done"
          onSubmitEditing={() => { addMutation.mutate(input); setInput(''); }} />
        <TouchableOpacity style={styles.addBtn} onPress={() => { addMutation.mutate(input); setInput(''); }}>
          <Text style={styles.addBtnText}>Add</Text>
        </TouchableOpacity>
      </View>
      <FlatList
        data={items}
        keyExtractor={i => i.id.toString()}
        contentContainerStyle={{ padding: 16 }}
        renderItem={({ item }) => (
          <View style={styles.item}>
            <Text style={styles.itemName}>{item.name}</Text>
            <Text style={styles.itemQty}>{item.qty}</Text>
          </View>
        )}
      />
    </View>
  );
}

  input:         { flex: 1, backgroundColor: 'white', borderRadius: 10, padding: 12, fontSize: 14,
