W9 Week 9 — Custom Hooks & Clean Architecture ▶

Refactor mixed-concern components into lean custom hooks. Each hook owns one responsibility; components just call hooks and render UI.

  1. 1 Take any component from your previous labs that has mixed concerns — state management, API calls, and business logic all tangled in one file.
  2. 2 Create a src/hooks/ folder in your project.
  3. 3 Extract useAuth.js — manages login/logout, token storage in AsyncStorage, exposes {user, login, logout, isLoading}.
  4. 4 Extract usePosts.js — fetches posts from API, exposes {posts, loading, error, refetch, addPost, deletePost}.
  5. 5 Extract useDebounce.js — function useDebounce(value, delay) — use useEffect + clearTimeout to delay state updates.
  6. 6 Each component now only calls hooks and renders UI — no direct API calls or AsyncStorage access inside components.
  7. 7 Add JSDoc comments to each hook documenting its parameters, return shape, and side effects.
  8. 8 Write unit tests for the pure logic in each hook using @testing-library/react-hooks.
→useAuth.js
/**
 * useAuth — manages authentication state and token storage.
 * @returns {Object} { user, login, logout, isLoading }
 */
import { useState, useEffect } from 'react';
import AsyncStorage from '@react-native-async-storage/async-storage';
import api from '../api/api';

export function useAuth() {
  const [user, setUser] = useState(null);
  const [isLoading, setIsLoading] = useState(true);

  useEffect(() => {
    (async () => {
      const raw = await AsyncStorage.getItem('user');
      if (raw) setUser(JSON.parse(raw));
      setIsLoading(false);
    })();
  }, []);

  /**
   * @param {{email: string, password: string}} credentials
   */
  const login = async (credentials) => {
    const { data } = await api.post('/auth/login', credentials);
    setUser(data.user);
    await AsyncStorage.setItem('user', JSON.stringify(data.user));
    await AsyncStorage.setItem('token', data.token);
  };

  const logout = async () => {
    setUser(null);
    await AsyncStorage.multiRemove(['user', 'token']);
  };

  return { user, login, logout, isLoading };
}
→usePosts.js
/**
 * usePosts — fetches and manages a list of posts.
 * @returns {Object} { posts, loading, error, refetch, addPost, deletePost }
 */
import { useState, useCallback } from 'react';
import api from '../api/api';

export function usePosts() {
  const [posts, setPosts] = useState([]);
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState(null);

  const refetch = useCallback(async () => {
    setLoading(true);
    setError(null);
    try {
      const { data } = await api.get('/posts');
      setPosts(data);
    } catch (err) {
      setError(err.message);
    } finally {
      setLoading(false);
    }
  }, []);

  /** @param {{title: string, body: string}} postData */
  const addPost = async (postData) => {
    const { data } = await api.post('/posts', postData);
    setPosts(prev => [data, ...prev]);
  };

  /** @param {number} id */
  const deletePost = async (id) => {
    await api.delete(`/posts/${id}`);
    setPosts(prev => prev.filter(p => p.id !== id));
  };

  return { posts, loading, error, refetch, addPost, deletePost };
}
→useDebounce.js
/**
 * useDebounce — delays a value update by a given delay.
 * @param {any} value   The value to debounce
 * @param {number} delay  Milliseconds to wait
 * @returns {any} debouncedValue
 */
import { useState, useEffect } from 'react';

export function useDebounce(value, delay = 400) {
  const [debouncedValue, setDebouncedValue] = useState(value);

  useEffect(() => {
    const timer = setTimeout(() => setDebouncedValue(value), delay);
    return () => clearTimeout(timer);   // cleanup on next keystroke
  }, [value, delay]);

  return debouncedValue;
}
W10 Week 10 — Writing Tests for React Native ▶

Build a real test suite: pure unit tests for reducers, hook tests, and component tests with @testing-library/react-native.

  1. 1 Set up Jest: Jest ships with Expo. Confirm it's configured by running npx jest --listTests from your project root.
  2. 2 Mock AsyncStorage: Add to your package.json Jest config: "setupFilesAfterEnv": ["@react-native-async-storage/async-storage/jest/async-storage-mock"]
  3. 3 Unit test — reducer: Test pure reducer functions with zero imports of React Native. 
// __tests__/todoReducer.test.js
import { todoReducer } from '../src/reducers/todoReducer';

test('ADD_TODO adds item to list', () => {
  const state = { todos: [] };
  const next = todoReducer(state, { type: 'ADD_TODO', payload: { id: 1, text: 'Buy milk' } });
  expect(next.todos).toHaveLength(1);
  expect(next.todos[0].text).toBe('Buy milk');
});

test('DELETE_TODO removes correct item', () => {
  const state = { todos: [{ id: 1, text: 'A' }, { id: 2, text: 'B' }] };
  const next = todoReducer(state, { type: 'DELETE_TODO', payload: 1 });
  expect(next.todos).toHaveLength(1);
  expect(next.todos[0].id).toBe(2);
});
  1. 4 Component test with @testing-library/react-native: 
// __tests__/TodoItem.test.js
import { render, fireEvent } from '@testing-library/react-native';
import TodoItem from '../src/components/TodoItem';

test('renders todo text', () => {
  const { getByText } = render(
    <TodoItem todo={{ id: 1, text: 'Buy milk', done: false }} onDelete={jest.fn()} />
  );
  expect(getByText('Buy milk')).toBeTruthy();
});

test('calls onDelete when delete pressed', () => {
  const mockDelete = jest.fn();
  const { getByTestId } = render(
    <TodoItem todo={{ id: 1, text: 'Buy milk', done: false }} onDelete={mockDelete} />
  );
  fireEvent.press(getByTestId('delete-btn'));
  expect(mockDelete).toHaveBeenCalledWith(1);
});
  1. 5 Run with coverage: npx jest --coverage
  2. 6 Target: 70%+ coverage on your hooks and reducer files. The coverage report will show uncovered branches in red — focus there first.
💡 If you see "Cannot find module 'react-native'" in tests, ensure your Jest preset is set to "preset": "jest-expo" in package.json.
W11 Week 11 — CI/CD for Mobile Apps ▶

Set up EAS Build to produce a real APK, then wire up GitHub Actions so every push runs lint and tests automatically.

  1. 1 Install EAS CLI globally: npm install -g eas-cli
  2. 2 Log in: eas login (create a free account at expo.dev if you don't have one)
  3. 3 Configure: eas build:configure — this creates eas.json with build profiles (development, preview, production)
  4. 4 Update app.json: set name, slug, version: "1.0.0", add icon and splash image paths
  5. 5 Build a preview APK for Android: eas build -p android --profile preview
  6. 6 Monitor the build at expo.dev/builds — takes 5–15 minutes on Expo's free tier
  7. 7 Download the APK → share via QR code with testers or send the direct download link
  8. 8 Create GitHub Actions CI workflow at .github/workflows/ci.yml: 
# .github/workflows/ci.yml
name: Mobile CI
on: [push, pull_request]

jobs:
  test:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v3
      - uses: actions/setup-node@v3
        with:
          node-version: '18'
      - run: npm ci
      - run: npx eslint src/ --ext .js,.jsx
      - run: npx jest --coverage --passWithNoTests
  1. 9 Push an OTA (Over-The-Air) update when you fix a JS bug — no app store review needed: eas update --branch production --message "Fix login bug"
  2. 10 OTA explained: EAS Update pushes a new JavaScript bundle to your users' installed app within seconds. Native code changes (new SDK version, new native module) still require a full EAS Build and store submission.
💡 For the GitHub Actions job to also trigger an EAS Build, add an EXPO_TOKEN secret to your GitHub repo (Settings → Secrets) and add eas build --non-interactive as a final step.
W12 Week 12 — Capstone: DevFeed Social App ▶

Build DevFeed — a developer community feed app demonstrating every skill from the programme. Full spec below.

→Screen Specifications
  1. 1SplashScreen: Show logo, check AsyncStorage for token, redirect to Login or Home automatically
  2. 2LoginScreen: Email + password fields, JWT auth API call, store token in AsyncStorage
  3. 3RegisterScreen: Name, email, password, confirm password — validate match before submit
  4. 4HomeScreen (TabBar): FlatList of posts with author avatar, text, timestamp, like count, comment count. Pull to refresh. Floating + button to create post
  5. 5CreatePostScreen (Stack): TextInput for text, TouchableOpacity to add image from gallery, submit to API
  6. 6PostDetailScreen: Full post + comments list + add comment TextInput at the bottom
  7. 7ProfileScreen (TabBar): User avatar, name, bio, their posts in a 3-column grid

→Navigation Setup — RootNavigator.js
// src/navigation/RootNavigator.js
import React from 'react';
import { NavigationContainer } from '@react-navigation/native';
import { createNativeStackNavigator } from '@react-navigation/native-stack';
import { createBottomTabNavigator } from '@react-navigation/bottom-tabs';
import { useAuth } from '../hooks/useAuth';

// Screens
import SplashScreen    from '../screens/SplashScreen';
import LoginScreen     from '../screens/LoginScreen';
import RegisterScreen  from '../screens/RegisterScreen';
import HomeScreen      from '../screens/HomeScreen';
import ProfileScreen   from '../screens/ProfileScreen';
import CreatePostScreen from '../screens/CreatePostScreen';
import PostDetailScreen from '../screens/PostDetailScreen';

const Stack = createNativeStackNavigator();
const Tab   = createBottomTabNavigator();

function MainTabs() {
  return (
    <Tab.Navigator screenOptions={{
      tabBarActiveTintColor: '#1B3A6B',
      tabBarStyle: { backgroundColor: '#fff', borderTopWidth: 1, borderTopColor: '#e5e7eb' },
      headerShown: false,
    }}>
      <Tab.Screen name="Home" component={HomeScreen}
        options={{ tabBarLabel: 'Feed', tabBarIcon: () => '🏠' }} />
      <Tab.Screen name="Profile" component={ProfileScreen}
        options={{ tabBarLabel: 'Profile', tabBarIcon: () => '👤' }} />
    </Tab.Navigator>
  );
}

export default function RootNavigator() {
  const { user, isLoading } = useAuth();
  if (isLoading) return <SplashScreen />;

  return (
    <NavigationContainer>
      <Stack.Navigator screenOptions={{ headerShown: false }}>
        {user ? (
          <>
            <Stack.Screen name="Main"       component={MainTabs} />
            <Stack.Screen name="CreatePost" component={CreatePostScreen}
              options={{ presentation: 'modal', headerShown: true, title: 'New Post' }} />
            <Stack.Screen name="PostDetail" component={PostDetailScreen}
              options={{ headerShown: true, title: 'Post' }} />
          </>
        ) : (
          <>
            <Stack.Screen name="Login"    component={LoginScreen} />
            <Stack.Screen name="Register" component={RegisterScreen} />
          </>
        )}
      </Stack.Navigator>
    </NavigationContainer>
  );
}
→api.js — Axios with Token Interceptor
// src/api/api.js
import axios from 'axios';
import AsyncStorage from '@react-native-async-storage/async-storage';

const api = axios.create({
  baseURL: 'https://api.devfeed.example.com/v1',
  timeout: 8000,
  headers: { 'Content-Type': 'application/json' },
});

// Attach token to every request
api.interceptors.request.use(async (config) => {
  const token = await AsyncStorage.getItem('token');
  if (token) config.headers.Authorization = `Bearer ${token}`;
  return config;
});

// Handle 401 globally — clear token and let auth state update
api.interceptors.response.use(
  (response) => response,
  async (error) => {
    if (error.response?.status === 401) {
      await AsyncStorage.multiRemove(['user', 'token']);
    }
    return Promise.reject(error);
  }
);

export default api;
→PostCard with Animated Like Button
// src/components/PostCard.js
import React, { useRef } from 'react';
import { View, Text, TouchableOpacity, Animated, StyleSheet } from 'react-native';
import Avatar from './Avatar';

export default function PostCard({ post, onLike }) {
  const scale = useRef(new Animated.Value(1)).current;

  const handleLike = () => {
    Animated.sequence([
      Animated.spring(scale, { toValue: 1.35, useNativeDriver: true, speed: 50 }),
      Animated.spring(scale, { toValue: 1,    useNativeDriver: true, speed: 30 }),
    ]).start();
    onLike(post.id);
  };

  return (
    <View style={styles.card}>
      <View style={styles.header}>
        <Avatar uri={post.author.avatar} size={38} />
        <View style={{ marginLeft: 10 }}>
          <Text style={styles.authorName}>{post.author.name}</Text>
          <Text style={styles.timestamp}>{post.createdAt}</Text>
        </View>
      </View>

      <Text style={styles.body}>{post.text}</Text>

      <View style={styles.footer}>
        <TouchableOpacity onPress={handleLike} style={styles.actionBtn}>
          <Animated.Text style={[styles.actionIcon, { transform: [{ scale }] }]}>
            {post.likedByMe ? '❤️' : '🤍'}
          </Animated.Text>
          <Text style={styles.actionLabel}>{post.likeCount}</Text>
        </TouchableOpacity>

        <TouchableOpacity style={styles.actionBtn}>
          <Text style={styles.actionIcon}>💬</Text>
          <Text style={styles.actionLabel}>{post.commentCount}</Text>
        </TouchableOpacity>
      </View>
    </View>
  );
}

const styles = StyleSheet.create({
  card:        { backgroundColor: '#fff', borderRadius: 12, padding: 16, marginBottom: 12,
                  shadowColor: '#000', shadowOffset: { width: 0, height: 2 },
                  shadowOpacity: 0.06, shadowRadius: 6, elevation: 2 },
  header:      { flexDirection: 'row', alignItems: 'center', marginBottom: 10 },
  authorName:  { fontWeight: '700', fontSize: 14, color: '#1B3A6B' },
  timestamp:   { fontSize: 11, color: '#9ca3af', marginTop: 2 },
  body:        { fontSize: 14, color: '#374151', lineHeight: 22, marginBottom: 12 },
  footer:      { flexDirection: 'row', gap: 20 },
  actionBtn:   { flexDirection: 'row', alignItems: 'center', gap: 4 },
  actionIcon:  { fontSize: 18 },
  actionLabel: { fontSize: 13, color: '#6b7280' },
});
