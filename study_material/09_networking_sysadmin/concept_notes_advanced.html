Week 9 Ansible & Configuration Management ▾

Ansible Architecture

Ansible is an agentless configuration management tool. It connects to managed nodes over SSH (or WinRM for Windows) from the control node and executes modules in Python.

ComponentDescription
Control NodeMachine where Ansible is installed and playbooks run from
Managed NodesTarget hosts — only need Python and SSH access (no agent)
InventoryList of managed nodes (hosts/groups); can be static file or dynamic script
PlaybookYAML file defining what to automate — contains Plays
PlayMaps a group of hosts to a list of tasks
TaskCalls a single Ansible module with arguments
ModuleUnit of work (apt, copy, service, template, file, user, etc.)
RoleReusable, structured collection of tasks, handlers, templates, and variables
HandlerTask triggered by notify — runs once at end of play (e.g., restart nginx)

Playbook Structure

---
# site.yml — full playbook example
- name: Configure Web Servers
  hosts: webservers
  become: yes                      # sudo for all tasks
  vars:
    nginx_port: 80
    app_user: "www-data"
  vars_files:
    - vars/secrets.yml             # Encrypted with Ansible Vault

  pre_tasks:
    - name: Update apt cache
      apt:
        update_cache: yes
        cache_valid_time: 3600

  tasks:
    - name: Install Nginx
      apt:
        name: nginx
        state: present             # present=install, absent=remove, latest=upgrade

    - name: Copy Nginx config
      template:
        src: templates/nginx.conf.j2
        dest: /etc/nginx/nginx.conf
        owner: root
        group: root
        mode: '0644'
      notify: Restart Nginx        # Triggers handler if file changed

    - name: Create app user
      user:
        name: "{{ app_user }}"
        system: yes
        shell: /bin/false

    - name: Ensure nginx is running and enabled
      service:
        name: nginx
        state: started
        enabled: yes

    - name: Run a shell command and capture output
      shell: nginx -v 2>&1
      register: nginx_version
      changed_when: false          # Shell tasks always report "changed" — override

    - name: Display nginx version
      debug:
        msg: "{{ nginx_version.stdout }}"

  handlers:
    - name: Restart Nginx
      service:
        name: nginx
        state: restarted

Key Modules Reference

# Package management
- apt:  name=nginx state=present update_cache=yes    # Debian/Ubuntu
- yum:  name=nginx state=present                     # RHEL/CentOS 7
- dnf:  name=nginx state=present                     # RHEL/CentOS 8+

# File operations
- copy:     src=files/app.conf dest=/etc/app.conf owner=root mode='0644'
- template: src=templates/app.conf.j2 dest=/etc/app.conf  # Jinja2 rendering
- file:     path=/data/logs state=directory owner=app mode='0755'
- file:     path=/tmp/old-file state=absent               # Delete file
- lineinfile: path=/etc/sshd_config regexp='^PermitRootLogin' line='PermitRootLogin no'

# Service management
- service:   name=nginx state=started enabled=yes
- systemd:   name=myapp state=restarted daemon_reload=yes

# User management
- user:   name=deploy groups=sudo,www-data shell=/bin/bash home=/home/deploy create_home=yes
- group:  name=developers state=present

# Git & archive
- git:    repo=https://github.com/myorg/app.git dest=/opt/app version=main
- unarchive: src=app.tar.gz dest=/opt/ remote_src=yes

# Command execution (avoid if idempotent module exists)
- command:  /usr/bin/python3 manage.py migrate  chdir=/opt/app
- shell:    source /opt/venv/bin/activate && pip install -r requirements.txt

HAProxy Configuration

# /etc/haproxy/haproxy.cfg

global
    maxconn 50000
    log /dev/log local0
    log /dev/log local1 notice
    user haproxy
    group haproxy
    daemon

defaults
    log global
    mode http                          # http mode for L7, tcp for L4
    option httplog
    option dontlognull
    option forwardfor                  # Add X-Forwarded-For header
    option http-server-close
    timeout connect 5s
    timeout client  30s
    timeout server  30s
    retries 3

frontend http-in
    bind *:80
    bind *:443 ssl crt /etc/ssl/haproxy/bundle.pem
    redirect scheme https if !{ ssl_fc }   # Redirect HTTP → HTTPS

    # ACL-based routing
    acl is_api path_beg /api/
    acl is_static path_end .css .js .png .jpg

    use_backend api-servers if is_api
    use_backend static-servers if is_static
    default_backend app-servers

backend app-servers
    balance roundrobin                 # roundrobin / leastconn / source (sticky)
    option httpchk GET /health HTTP/1.1\r\nHost:\ app.example.com
    http-check expect status 200
    server app1 192.168.1.10:8080 check inter 2s fall 3 rise 2 maxconn 300
    server app2 192.168.1.11:8080 check inter 2s fall 3 rise 2 maxconn 300
    server app3 192.168.1.12:8080 check inter 2s fall 3 rise 2 maxconn 300 backup  # Backup only

backend api-servers
    balance leastconn
    server api1 192.168.1.20:8000 check
    server api2 192.168.1.21:8000 check

# Stats page
listen stats
    bind *:8404
    stats enable
    stats uri /haproxy-stats
    stats auth admin:StrongPassword123
    stats refresh 5s
    stats show-legends

HAProxy Load Balancing Algorithms

AlgorithmDescriptionUse Case
roundrobinEach server in turn (respects weights)Stateless apps, equal server capacity
leastconnSend to server with fewest active connectionsLong-lived connections (DB, WebSocket)
sourceHash source IP → same server (sticky)Simple session persistence without cookies
uriHash request URI → consistent serverCaching servers, maximise cache hits
firstFill first server to maxconn, then nextMinimise servers running (cost optimization)

keepalived — VRRP Virtual IP

# /etc/keepalived/keepalived.conf on PRIMARY (lb1 - 192.168.1.10)
vrrp_script chk_haproxy {
    script "/usr/bin/killall -0 haproxy"   # Check if haproxy is running
    interval 2
    weight -20                             # Reduce priority by 20 if check fails
}

vrrp_instance VI_1 {
    state MASTER                           # MASTER on primary, BACKUP on secondary
    interface eth0
    virtual_router_id 51                   # Must match on both nodes
    priority 110                           # MASTER has higher priority than BACKUP (100)
    advert_int 1                           # Send VRRP advertisement every 1 second
    authentication {
        auth_type PASS
        auth_pass SecureVRRP123
    }
    virtual_ipaddress {
        192.168.1.100/24 dev eth0          # VIP — clients connect to this IP
    }
    track_script {
        chk_haproxy
    }
    notify_master "/etc/keepalived/notify.sh MASTER"
    notify_backup "/etc/keepalived/notify.sh BACKUP"
}

# On BACKUP node (lb2 - 192.168.1.11):
# state BACKUP, priority 100 (lower than MASTER)
# All other settings identical

Nginx Load Balancing

# /etc/nginx/nginx.conf — upstream load balancing
http {
    upstream app_backend {
        least_conn;                         # least_conn / ip_hash / (default=round-robin)
        server 192.168.1.10:8080 weight=3;  # Receives 3x more traffic
        server 192.168.1.11:8080 weight=1;
        server 192.168.1.12:8080 down;      # Marked down (out of rotation)
        server 192.168.1.13:8080 backup;    # Only used when others are down
        keepalive 32;                       # Keep 32 connections to each backend open
    }

    server {
        listen 80;
        server_name app.example.com;

        location / {
            proxy_pass http://app_backend;
            proxy_set_header Host $host;
            proxy_set_header X-Real-IP $remote_addr;
            proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
            proxy_connect_timeout 5s;
            proxy_read_timeout 30s;
            proxy_next_upstream error timeout invalid_header http_500;
        }

        location /health {
            return 200 "OK\n";
            add_header Content-Type text/plain;
        }
    }
}

BGP Anycast & DNS-Based LB

# BGP Anycast: same IP prefix advertised from multiple PoPs
# Traffic routed to nearest PoP by BGP best-path selection
# Used by: Cloudflare (1.1.1.1), Google (8.8.8.8), root DNS servers
# Pros: very fast failover (BGP reconvergence), geo proximity
# Cons: requires ISP relationships/BGP capable infrastructure

# DNS Round-Robin LB:
; DNS zone file — multiple A records for same name
app.example.com.  60  IN  A  203.0.113.10
app.example.com.  60  IN  A  203.0.113.11
app.example.com.  60  IN  A  203.0.113.12
# Low TTL (60s) means failures are detected quickly
# But: no health checks — remove failed IPs manually or via automation

# GSLB (Global Server Load Balancing) via Route 53 Health Checks:
# Primary: us-east-1 ALB
# Secondary: eu-west-1 ALB
# Route 53 health check → if primary fails, DNS switches to secondary
# Failover routing policy with health checks achieves automatic DR
Week 11 AWS Networking & Cloud Infrastructure ▾

VPC — Virtual Private Cloud

# VPC = your isolated network environment in AWS
# Spans all Availability Zones in a region
# CIDR block: e.g., 10.0.0.0/16 (65,534 usable IPs)

VPC (10.0.0.0/16)
├── Public Subnet AZ-a  (10.0.1.0/24)  → Internet Gateway route
├── Public Subnet AZ-b  (10.0.2.0/24)  → Internet Gateway route
├── Private Subnet AZ-a (10.0.10.0/24) → NAT Gateway route
├── Private Subnet AZ-b (10.0.11.0/24) → NAT Gateway route
├── Database Subnet AZ-a (10.0.20.0/24) → No internet route
└── Database Subnet AZ-b (10.0.21.0/24) → No internet route

# Internet Gateway (IGW):
# - One per VPC, enables public internet access
# - Attach to VPC, then add route 0.0.0.0/0 → igw-xxx in public subnet route table

# NAT Gateway:
# - Placed in PUBLIC subnet, assigned Elastic IP
# - Private subnets route 0.0.0.0/0 → nat-xxx
# - Enables outbound internet (e.g., apt update) from private instances
# - Inbound connections NOT allowed (stateful, but no inbound NAT)
# - Cost: ~$0.045/hr + $0.045/GB data processed

VPC Connectivity Options

# VPC Peering:
# - Direct connection between two VPCs (same or cross-account/region)
# - Requires route table entries on both sides
# - NOT transitive: A↔B, B↔C does NOT mean A can reach C
# - Use case: connect app VPC to monitoring VPC

# Transit Gateway (TGW):
# - AWS-managed hub: connects many VPCs + on-prem (via VPN or Direct Connect)
# - Transitive routing supported
# - Centralised routing policies and security
# - Use case: large multi-VPC architectures (replace full mesh of peering)

# VPN (Site-to-Site):
# - IPSec tunnel between on-prem router and AWS Virtual Private Gateway
# - Up to 1.25 Gbps per tunnel, multiple tunnels for redundancy

# Direct Connect:
# - Dedicated physical connection (1G/10G/100G) to AWS
# - Consistent bandwidth and latency (no internet variability)
# - Requires co-location at Direct Connect location or partner
# - Use case: hybrid cloud, large data transfers, compliance requirements

# VPC Endpoints:
# - Access AWS services (S3, DynamoDB, SSM) without internet
# - Interface endpoint (PrivateLink): ENI in your VPC, uses SG
# - Gateway endpoint: S3 and DynamoDB only, route table entry (free)

CloudFront CDN

# CloudFront distributes content from 400+ edge locations globally
# Origin → CloudFront Edge → User (cache reduces latency + origin load)

# Key concepts:
# Distribution: your CloudFront configuration
# Origin:       S3 bucket, ALB, EC2, or custom HTTP server
# Behavior:     rules matching URL paths to origins + cache settings
# Cache Policy: what headers/cookies/query strings to include in cache key
# OAC:          Origin Access Control — allows CloudFront to access private S3

# Cache headers:
Cache-Control: max-age=31536000    # Cache for 1 year (static assets)
Cache-Control: no-cache            # Revalidate every time
Cache-Control: s-maxage=3600       # CDN-specific TTL

# Lambda@Edge:
# Run Node.js or Python at edge locations
# Use cases: A/B testing, HTTP redirects, auth at edge, personalisation
# Triggers: viewer request/response, origin request/response

# CloudFront functions:
# Lighter weight (JavaScript only), only viewer-side events, microseconds execution

AWS Global Accelerator

# Global Accelerator provides two static anycast IPs
# Traffic enters AWS backbone at nearest edge PoP (vs. internet routing)
# Routes to optimal endpoint (ALB/NLB/EC2) in multiple regions
# Provides instant failover (<30s) across regions
# Improves TCP connection performance (backbone vs public internet)

# Comparison: CloudFront vs Global Accelerator:
# CloudFront:          HTTP/HTTPS content caching, cache at edge
# Global Accelerator:  Any protocol (TCP/UDP), routes to healthy endpoints, no caching
Week 12 Kubernetes Networking & Terraform IaC ▾

Kubernetes Network Model

Kubernetes enforces a flat, routable network model with three rules:

  1. Every Pod gets a unique cluster-wide IP address
  2. Pods can communicate with any other Pod without NAT
  3. Agents on a Node can communicate with all Pods on that Node
# Pod networking:
# - Each Pod gets IP from pod CIDR (e.g., 10.244.0.0/16)
# - Pod IP is ephemeral — changes when Pod restarts
# - Services provide stable virtual IPs (ClusterIP from service CIDR)

# CNI Plugins:
# Flannel:  simple overlay using VXLAN; easy setup, no NetworkPolicy
# Calico:   L3 routing (BGP or IPIP), NetworkPolicy, high performance
# Cilium:   eBPF-based; L7 NetworkPolicy, Hubble observability, zero overhead
# Weave:    encrypted mesh, simple, works in complex topologies
# Canal:    Flannel for networking + Calico for NetworkPolicy

apiVersion: networking.k8s.io/v1
kind: NetworkPolicy
metadata:
  name: allow-api-to-db
  namespace: production
spec:
  podSelector:
    matchLabels:
      role: database               # Apply to database pods
  policyTypes:
    - Ingress
  ingress:
    - from:
        - podSelector:
            matchLabels:
              role: api            # Only allow API pods to reach database
      ports:
        - protocol: TCP
          port: 5432

# kube-prometheus-stack (Helm chart) — installs everything in one go
helm repo add prometheus-community https://prometheus-community.github.io/helm-charts
helm install kube-prom prometheus-community/kube-prometheus-stack \
  --namespace monitoring --create-namespace \
  --set grafana.adminPassword=SecurePass123 \
  --set prometheus.prometheusSpec.retention=30d

# Components installed:
# - Prometheus Operator + Prometheus instance
# - Grafana (dashboards pre-loaded)
# - Alertmanager
# - node-exporter (DaemonSet — metrics from every node)
# - kube-state-metrics (Deployment state, replica counts, etc.)
# - cAdvisor (container CPU/memory metrics, built into kubelet)

# Key Kubernetes metrics:
# kube_pod_status_phase{phase="Running"}         — running pod count
# kube_deployment_spec_replicas                  — desired replicas
# kube_deployment_status_replicas_available      — available replicas
# container_cpu_usage_seconds_total              — container CPU usage
# container_memory_working_set_bytes             — container memory (what OOM uses)

# HPA — Horizontal Pod Autoscaler
apiVersion: autoscaling/v2
kind: HorizontalPodAutoscaler
metadata:
  name: api-hpa
spec:
  scaleTargetRef:
    apiVersion: apps/v1
    kind: Deployment
    name: api-deployment
  minReplicas: 2
  maxReplicas: 10
  metrics:
    - type: Resource
      resource:
        name: cpu
        target:
          type: Utilization
          averageUtilization: 70   # Scale up when avg CPU > 70%
