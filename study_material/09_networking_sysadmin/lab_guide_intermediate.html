Week 5 Lab Firewall Hardening & WireGuard VPN ▾

Objectives

  • Implement a production-grade UFW firewall policy with source restrictions
  • Create persistent iptables rules with iptables-persistent
  • Set up a WireGuard VPN tunnel between server1 and server2
  • Configure fail2ban to protect against brute-force SSH attacks

Part A — UFW Firewall Configuration

# Start with a clean slate — reset any existing UFW rules
sudo ufw reset
sudo ufw status

# Set default policies (deny all inbound, allow all outbound)
sudo ufw default deny incoming
sudo ufw default allow outgoing

# Allow essential services with comments for documentation
sudo ufw allow 22/tcp comment 'SSH access'
sudo ufw allow 80/tcp comment 'HTTP web server'
sudo ufw allow 443/tcp comment 'HTTPS web server'

# Restrict MySQL to internal network only (never expose DB to internet)
sudo ufw allow from 192.168.56.0/24 to any port 3306 comment 'MySQL - internal LAN only'

# Allow ICMP (ping) — this is handled by /etc/ufw/before.rules by default
# Verify: grep icmp /etc/ufw/before.rules | head -5

# Enable the firewall
sudo ufw enable
# Type 'y' when prompted

# Verify the rules
sudo ufw status verbose
# Expected output of 'sudo ufw status verbose':
# Status: active
# Logging: on (low)
# Default: deny (incoming), allow (outgoing), disabled (routed)
# New profiles: skip
#
# To                         Action      From
# --                         ------      ----
# 22/tcp                     ALLOW IN    Anywhere            # SSH access
# 80/tcp                     ALLOW IN    Anywhere            # HTTP web server
# 443/tcp                    ALLOW IN    Anywhere            # HTTPS web server
# 3306                       ALLOW IN    192.168.56.0/24     # MySQL - internal LAN only

Part B — Persistent iptables Rules

# Install iptables-persistent to save/restore rules across reboots
sudo apt install iptables-persistent -y
# Answer YES to save current rules when prompted

# Create a clean ruleset file
sudo tee /etc/iptables/rules.v4 <<'EOF'
*filter
# Default policies
:INPUT DROP [0:0]
:FORWARD DROP [0:0]
:OUTPUT ACCEPT [0:0]

# Allow loopback (local traffic — always needed)
-A INPUT -i lo -j ACCEPT
-A OUTPUT -o lo -j ACCEPT

# Allow established and related connections (stateful)
-A INPUT -m state --state ESTABLISHED,RELATED -j ACCEPT

# SSH — allow from anywhere (restrict to specific IPs in production)
-A INPUT -p tcp --dport 22 -j ACCEPT

# HTTP and HTTPS
-A INPUT -p tcp --dport 80 -j ACCEPT
-A INPUT -p tcp --dport 443 -j ACCEPT

# ICMP (ping) — allow all types
-A INPUT -p icmp -j ACCEPT

# MySQL — only from internal network
-A INPUT -p tcp --dport 3306 -s 192.168.56.0/24 -j ACCEPT

# Log dropped packets (before final drop)
-A INPUT -m limit --limit 5/min -j LOG --log-prefix "iptables-drop: " --log-level 7

COMMIT
EOF

# Apply the rules
sudo iptables-restore < /etc/iptables/rules.v4

# Verify rules are active
sudo iptables -L INPUT -n -v
# Expected: Shows all defined rules with packet counters

Part C — WireGuard VPN Setup

# Install WireGuard on BOTH servers
sudo apt install wireguard -y

# ===== ON SERVER1 (192.168.56.10) =====

# Generate key pair
wg genkey | sudo tee /etc/wireguard/server1_private.key | wg pubkey | sudo tee /etc/wireguard/server1_public.key
sudo chmod 600 /etc/wireguard/server1_private.key

# Read and save the keys
SERVER1_PRIVATE=$(sudo cat /etc/wireguard/server1_private.key)
SERVER1_PUBLIC=$(sudo cat /etc/wireguard/server1_public.key)
echo "Server1 Public Key: $SERVER1_PUBLIC"
# COPY THIS KEY — you need it for server2's config
# ===== ON SERVER2 (192.168.56.11) =====

# Generate key pair on server2
wg genkey | sudo tee /etc/wireguard/server2_private.key | wg pubkey | sudo tee /etc/wireguard/server2_public.key
sudo chmod 600 /etc/wireguard/server2_private.key

SERVER2_PRIVATE=$(sudo cat /etc/wireguard/server2_private.key)
SERVER2_PUBLIC=$(sudo cat /etc/wireguard/server2_public.key)
echo "Server2 Public Key: $SERVER2_PUBLIC"
# COPY THIS KEY — you need it for server1's config
# ===== BACK ON SERVER1 — Create wg0.conf =====
sudo tee /etc/wireguard/wg0.conf <<EOF
[Interface]
PrivateKey = $SERVER1_PRIVATE
Address = 10.0.0.1/24
ListenPort = 51820

# Enable IP forwarding through the VPN (optional — for routing lab traffic)

[Peer]
# Server2's public key
PublicKey = <PASTE_SERVER2_PUBLIC_KEY_HERE>
AllowedIPs = 10.0.0.2/32
Endpoint = 192.168.56.11:51820
PersistentKeepalive = 25
EOF
sudo chmod 600 /etc/wireguard/wg0.conf
# ===== ON SERVER2 — Create wg0.conf =====
sudo tee /etc/wireguard/wg0.conf <<EOF
[Interface]
PrivateKey = $SERVER2_PRIVATE
Address = 10.0.0.2/24

[Peer]
# Server1's public key

PersistentKeepalive = 25

Part D — fail2ban SSH Protection

# Install fail2ban
sudo apt install fail2ban -y

# Create local jail configuration (never edit jail.conf — it gets overwritten on upgrade)
sudo tee /etc/fail2ban/jail.local <<'EOF'
[DEFAULT]
# Global defaults
bantime  = 3600       # Ban duration: 1 hour
findtime = 600        # Count failures within this window: 10 minutes
maxretry = 3          # Attempts before ban
ignoreip = 127.0.0.1/8 ::1 192.168.56.0/24  # Never ban these
banaction = iptables-multiport
backend = systemd     # Use systemd journal for log parsing

[sshd]
enabled  = true
port     = 22
logpath  = %(sshd_log)s
maxretry = 3

[nginx-http-auth]
enabled  = true
EOF

# Start fail2ban
sudo systemctl enable --now fail2ban
sudo systemctl status fail2ban

# Verify SSH jail is active
sudo fail2ban-client status
# Expected: "Number of jail:   2"
# "Jail list:   nginx-http-auth, sshd"

sudo fail2ban-client status sshd
# Expected: Shows currently banned IPs (likely empty)
># Test fail2ban by simulating failed SSH logins
# From server2, attempt 4 failed logins (enter wrong password each time):
for i in 1 2 3 4; do
    ssh -o ConnectTimeout=5 wronguser@192.168.56.10 || true
    sleep 1
done

# Back on server1, check if server2 is now banned
sudo fail2ban-client status sshd
# Expected: "Banned IP list: 192.168.56.11"

# Unban server2 (so you can continue the labs!)
sudo fail2ban-client set sshd unbanip 192.168.56.11
sudo fail2ban-client status sshd   # Should show empty ban list

Part A — Custom Python Web Server as a systemd Service

# Create the web server root directory
sudo mkdir -p /var/www/html
echo "<h1>Lab Web Server</h1><p>Served by Python on port 8080</p>" | sudo tee /var/www/html/index.html

# Create the Python web server script
sudo tee /usr/local/bin/webserver.py <<'EOF'
#!/usr/bin/env python3
"""Simple HTTP server for lab use"""
from http.server import HTTPServer, SimpleHTTPRequestHandler
import os
import logging

logging.basicConfig(
    level=logging.INFO,
    format='%(asctime)s - %(levelname)s - %(message)s'
)

os.chdir('/var/www/html')

class LoggingHandler(SimpleHTTPRequestHandler):
    def log_message(self, format, *args):
        logging.info("%s - %s" % (self.client_address[0], format % args))

PORT = 8080
httpd = HTTPServer(('0.0.0.0', PORT), LoggingHandler)
logging.info(f"Starting web server on port {PORT}")
httpd.serve_forever()
EOF

sudo chmod +x /usr/local/bin/webserver.py
# Create the systemd unit file
sudo tee /etc/systemd/system/webserver.service <<'EOF'
[Unit]
Description=Simple Python Web Server
Documentation=https://docs.python.org/3/library/http.server.html
After=network.target
Wants=network-online.target

[Service]
Type=simple
User=www-data
Group=www-data

# Working directory and command
WorkingDirectory=/var/www/html
ExecStart=/usr/bin/python3 /usr/local/bin/webserver.py

# Restart policy
Restart=always
RestartSec=5
StartLimitBurst=3
StartLimitIntervalSec=60

# Resource limits
LimitNOFILE=65536

# Logging
StandardOutput=journal
StandardError=journal
SyslogIdentifier=webserver

[Install]
WantedBy=multi-user.target
EOF
# Reload systemd, enable and start the service
sudo systemctl daemon-reload
sudo systemctl enable webserver
sudo systemctl start webserver

# Verify it's running
sudo systemctl status webserver
# Expected: active (running)

# Test the web server
curl http://localhost:8080
# Expected: <h1>Lab Web Server</h1>

# Allow port 8080 through firewall
sudo ufw allow 8080/tcp comment 'Python lab web server'

# Test from server2
curl http://192.168.56.10:8080/
# Expected: HTML from server1

# View logs
sudo journalctl -u webserver -n 20
sudo journalctl -u webserver -f    # Follow live (Ctrl+C to stop)

Part B — Service Watchdog with cron

# Create the watchdog script
sudo tee /usr/local/bin/service_watchdog.sh <<'EOF'
#!/bin/bash
# Service watchdog — restarts service if it goes down
SERVICE="webserver"
LOG="/var/log/watchdog.log"
TIMESTAMP=$(date '+%Y-%m-%d %H:%M:%S')

if ! systemctl is-active --quiet "$SERVICE"; then
    echo "$TIMESTAMP: WARNING - $SERVICE is DOWN, attempting restart..." | tee -a "$LOG"
    systemctl restart "$SERVICE"
    sleep 2

    if systemctl is-active --quiet "$SERVICE"; then
        echo "$TIMESTAMP: SUCCESS - $SERVICE restarted successfully" | tee -a "$LOG"
        logger -t watchdog "Service $SERVICE was restarted successfully"
    else
        echo "$TIMESTAMP: ERROR - $SERVICE failed to restart!" | tee -a "$LOG"
        logger -p user.err -t watchdog "CRITICAL: Service $SERVICE failed to restart"
    fi
else
    echo "$TIMESTAMP: OK - $SERVICE is running" >> "$LOG"
fi
EOF

sudo chmod +x /usr/local/bin/service_watchdog.sh
sudo /usr/local/bin/service_watchdog.sh   # Test it runs ok
cat /var/log/watchdog.log
# Schedule watchdog in cron (runs every 5 minutes)
(crontab -l 2>/dev/null; echo "*/5 * * * * /usr/local/bin/service_watchdog.sh") | crontab -
crontab -l   # Verify entry added

# Test the watchdog — stop the service and wait for cron to restart it
sudo systemctl stop webserver
curl http://localhost:8080   # Should fail
sleep 310                    # Wait 5+ minutes (or manually run the script)
sudo /usr/local/bin/service_watchdog.sh   # Simulate cron run
curl http://localhost:8080   # Should work again
cat /var/log/watchdog.log    # Should show WARNING and SUCCESS lines

# FIRST: Add a new virtual disk in VirtualBox
# 1. Power off server1
# 2. VirtualBox → server1 Settings → Storage

lsblk
# Expected: /dev/sdb   10G (new disk, unformatted)
# Initialize LVM
# Step 1: Create Physical Volume (PV) on the new disk
sudo pvcreate /dev/sdb
sudo pvdisplay            # Verify PV created

# Step 2: Create Volume Group (VG) named 'datavg'
sudo vgcreate datavg /dev/sdb
sudo vgdisplay            # Verify VG, note Free PE count

# Step 3: Create Logical Volume (LV) named 'datalv' — 8GB
sudo lvcreate -L 8G -n datalv datavg
sudo lvdisplay            # Verify LV, note device path

# Step 4: Format with ext4
sudo mkfs.ext4 /dev/datavg/datalv

# Step 5: Mount it
sudo mkdir -p /data2
sudo mount /dev/datavg/datalv /data2
df -h /data2
# Expected: /dev/mapper/datavg-datalv  7.9G  36M  7.5G   1%  /data2
# Step 6: Make mount persistent via /etc/fstab
echo '/dev/datavg/datalv  /data2  ext4  defaults  0  2' | sudo tee -a /etc/fstab
# Verify fstab syntax
sudo mount -a    # Re-mount all from fstab (should not error)
df -h /data2     # Should still show mounted
# Step 7: Extend the LV (add 1GB — simulate growing disk needs)
sudo lvextend -L +1G /dev/datavg/datalv
sudo lvdisplay | grep "LV Size"    # Should now show 9.00 GiB

# Extend the filesystem (online — no unmount needed)
sudo resize2fs /dev/datavg/datalv
df -h /data2                       # Should now show ~8.8G total

# Summary commands
sudo pvs; sudo vgs; sudo lvs       # Compact overview
sudo pvdisplay; sudo vgdisplay; sudo lvdisplay   # Full details

Part D — logrotate Configuration

# Create some test log files
sudo mkdir -p /var/log/webserver

# Create logrotate config for the webserver logs

/var/log/webserver/*.log {
    daily
    rotate 7
    compress
    delaycompress
    missingok
    notifempty
    create 0640 www-data adm
    sharedscripts

    endscript
}
EOF

[Journal]
SystemMaxUse=500M
MaxRetentionSec=1month

EOF

Part D — Alerting Rules

># Create alert rules file
sudo tee /etc/prometheus/alerts.yml <<'EOF'
groups:
  - name: node_alerts
    interval: 30s
    rules:

      - alert: HighCPUUsage
        expr: |
          100 - (avg by(instance) (
            rate(node_cpu_seconds_total{mode="idle"}[5m])
          ) * 100) > 80
        for: 5m
        labels:
          severity: warning
          team: devops
        annotations:
          summary: "High CPU usage on {{ $labels.instance }}"
          description: "CPU usage is {{ printf \"%.1f\" $value }}% (threshold: 80%)"

      - alert: HighMemoryUsage
        expr: |
          (node_memory_MemTotal_bytes - node_memory_MemAvailable_bytes)
          / node_memory_MemTotal_bytes * 100 > 85
        for: 5m
        labels:
          severity: warning
        annotations:
          summary: "High memory on {{ $labels.instance }}"
          description: "Memory usage {{ printf \"%.1f\" $value }}%"

      - alert: DiskSpaceLow
        expr: |
          node_filesystem_avail_bytes{mountpoint="/", fstype!="tmpfs"}
          / node_filesystem_size_bytes{mountpoint="/", fstype!="tmpfs"}
          * 100 < 20
        for: 5m
        labels:
          severity: critical
        annotations:
          summary: "Disk space low on {{ $labels.instance }}"
          description: "Disk {{ $labels.mountpoint }} has only {{ printf \"%.1f\" $value }}% free"

      - alert: NodeDown
        expr: up{job="node"} == 0
        for: 1m
        labels:
          severity: critical
        annotations:
          summary: "Node is down: {{ $labels.instance }}"
          description: "Node exporter on {{ $labels.instance }} is unreachable"

      - alert: ServiceNotRunning
        expr: node_systemd_unit_state{state="active", name=~"nginx.service|webserver.service"} == 0
        for: 2m
        labels:
          severity: warning
        annotations:
          summary: "Service {{ $labels.name }} is not active"
EOF

sudo chown prometheus:prometheus /etc/prometheus/alerts.yml

# Validate rules
promtool check rules /etc/prometheus/alerts.yml
# Expected: "SUCCESS: 5 rules found"

# Reload Prometheus to pick up the new rules
sudo systemctl reload prometheus
# Check: http://192.168.56.10:9090/alerts — should list all 5 alerts
