Quick Reference — Database Administration

SQL patterns, JOIN examples, window functions, transactions, index cheatsheet, MySQL vs PostgreSQL comparison, and MongoDB/Redis snippets — all in one printable page.

1. SQL Fundamentals

-- DDL
CREATE DATABASE mydb CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;
USE mydb;

CREATE TABLE users (
  id         INT PRIMARY KEY AUTO_INCREMENT,
  name       VARCHAR(100) NOT NULL,
  email      VARCHAR(150) NOT NULL UNIQUE,
  role       ENUM('admin','user','guest') DEFAULT 'user',
  created_at DATETIME DEFAULT CURRENT_TIMESTAMP
);

ALTER TABLE users ADD COLUMN last_login DATETIME;
CREATE INDEX idx_email ON users(email);
DROP TABLE IF EXISTS temp_users;

-- DML
INSERT INTO users (name, email) VALUES ('Priya', 'priya@example.com');
INSERT INTO users (name, email) VALUES ('Rahul','r@e.com'),('Ankit','a@e.com');

SELECT id, name, email
FROM users
WHERE role = 'user'
ORDER BY created_at DESC
LIMIT 10 OFFSET 20;

UPDATE users SET role = 'admin' WHERE email = 'priya@example.com';
DELETE FROM users WHERE last_login < DATE_SUB(NOW(), INTERVAL 1 YEAR);

-- Aggregate
SELECT city, COUNT(*) AS cnt, AVG(age) AS avg_age
FROM users
GROUP BY city
HAVING cnt > 5
ORDER BY cnt DESC;

2. JOIN Patterns

-- INNER JOIN — matching rows only
SELECT o.id, c.name, o.total
FROM orders o
JOIN customers c ON o.customer_id = c.id
WHERE o.status = 'delivered';

-- LEFT JOIN — all customers even with 0 orders
SELECT c.name, COUNT(o.id) AS order_count
FROM customers c
LEFT JOIN orders o ON c.id = o.customer_id
GROUP BY c.id, c.name;

-- Find customers with NO orders
SELECT c.name FROM customers c
LEFT JOIN orders o ON c.id = o.customer_id
WHERE o.id IS NULL;

-- Multi-table JOIN
SELECT c.name, p.name AS product, oi.quantity, oi.unit_price
FROM customers c
JOIN orders o      ON c.id          = o.customer_id
JOIN order_items oi ON o.id          = oi.order_id
JOIN products p    ON oi.product_id  = p.id
WHERE o.created_at >= '2024-01-01';

-- Self JOIN (employee → manager)
SELECT e.name AS employee, m.name AS manager
FROM employees e
LEFT JOIN employees m ON e.manager_id = m.id;

3. Window Functions (MySQL 8+ / PostgreSQL)

-- ROW_NUMBER, RANK, DENSE_RANK
SELECT name, salary,
  ROW_NUMBER()  OVER (PARTITION BY dept ORDER BY salary DESC) AS row_num,
  RANK()        OVER (PARTITION BY dept ORDER BY salary DESC) AS rnk,
  DENSE_RANK()  OVER (PARTITION BY dept ORDER BY salary DESC) AS dense_rnk
FROM employees;
-- ROW_NUMBER: 1,2,3,4    — always unique
-- RANK:       1,2,2,4    — gap after tie
-- DENSE_RANK: 1,2,2,3    — no gap after tie

-- Running total and previous value
SELECT date, amount,
  SUM(amount) OVER (ORDER BY date ROWS UNBOUNDED PRECEDING) AS running_total,
  LAG(amount, 1, 0)  OVER (ORDER BY date) AS prev_amount,
  LEAD(amount, 1, 0) OVER (ORDER BY date) AS next_amount
FROM daily_sales;

-- NTILE — quartiles
SELECT name, score,
  NTILE(4) OVER (ORDER BY score DESC) AS quartile
FROM students;

4. Transactions

START TRANSACTION;
  -- Step 1: deduct stock
  UPDATE products SET stock = stock - 2 WHERE id = 5;
  -- Step 2: create order
  INSERT INTO orders (customer_id, total) VALUES (10, 499.98);
  -- Step 3: order item
  INSERT INTO order_items (order_id, product_id, quantity)
  VALUES (LAST_INSERT_ID(), 5, 2);
COMMIT;
-- On any error: ROLLBACK;

-- Savepoint (partial rollback)
START TRANSACTION;
  INSERT INTO t VALUES (1);
  SAVEPOINT sp1;
  INSERT INTO t VALUES (2);
  ROLLBACK TO sp1;   -- (2) reverted; (1) remains
COMMIT;

-- Explicit locking
SELECT * FROM products WHERE id = 5 FOR UPDATE;  -- pessimistic lock

5. Indexes & EXPLAIN

-- Create indexes
CREATE INDEX idx_status           ON orders(status);
CREATE INDEX idx_composite        ON order_items(order_id, product_id);
CREATE UNIQUE INDEX idx_email     ON users(email);
DROP INDEX idx_status ON orders;
SHOW INDEXES FROM orders;

-- Read EXPLAIN output
EXPLAIN SELECT * FROM orders WHERE customer_id = 42;
-- type=ALL  → full scan (bad for large tables)
-- type=ref  → index lookup (good)
-- type=const → primary key lookup (best)
-- key=NULL  → no index used
-- rows=10000 → estimated rows scanned (lower = faster)
-- Extra: Using filesort → ORDER BY without index (add index)
-- Extra: Using index   → covering index (fastest)

-- Avoid these anti-patterns:
-- ✗ WHERE YEAR(created_at) = 2024     (function on indexed col)
-- ✓ WHERE created_at BETWEEN '2024-01-01' AND '2024-12-31'
-- ✗ WHERE email LIKE '%gmail.com'     (leading wildcard)
-- ✓ WHERE email LIKE 'priya%'         (trailing wildcard uses index)

7. MongoDB + Redis Quick Reference

// MongoDB (mongoose)
const Post = mongoose.model('Post', schema);

// Query: posts with tag 'node' and views > 100
await Post.find({ tags: 'node', views: { $gt: 100 } }).sort('-views').limit(5);

// Aggregation: top 3 authors by total views
await Post.aggregate([
  { $group: { _id: '$author', total: { $sum: '$views' } } },
  { $sort: { total: -1 } },
  { $limit: 3 }
]);

// Update: increment views, push to array
await Post.updateOne({ _id: id }, {
  $inc: { views: 1 },
  $push: { comments: { user, text, createdAt: new Date() } }
});

// Delete: older than 2020
await Post.deleteMany({ createdAt: { $lt: new Date('2020-01-01') } });

// ────────────────────────────────────────────────────────
// Redis (ioredis)
const redis = new Redis();

// String with TTL
await redis.set('key', 'value', 'EX', 300);       // 5-min TTL
await redis.setex('key', 300, 'value');            // equivalent
const val = await redis.get('key');

// Hash (object storage)
await redis.hset('user:42', 'name', 'Priya', 'role', 'admin');
const user = await redis.hgetall('user:42');
await redis.expire('user:42', 3600);               // 1-hour TTL

// Sorted set (leaderboard)
await redis.zadd('leaderboard', 1500, 'alice', 1200, 'bob');
const top5 = await redis.zrevrange('leaderboard', 0, 4, 'WITHSCORES');
await redis.zincrby('leaderboard', 100, 'bob');    // add 100 to bob's score

// Rate limiting
await redis.incr(`requests:${ip}:minute`);         // atomic counter
await redis.expire(`requests:${ip}:minute`, 60);   // 1-min window

// Cache-aside pattern
const cached = await redis.get(cacheKey);
if (cached) return JSON.parse(cached);
const data = await db.query(...);
await redis.setex(cacheKey, 300, JSON.stringify(data));
