Lab 1

Database Design — Library Management System (Week 5)

▼

Objective

Design a full library management system from ER diagram through to complex analytical queries.

Step-by-Step

  1. Design the schema
    Key design decisions:
    • Use VARCHAR(20) for ISBN — not INT — because ISBNs contain dashes (978-0-596-51774-8)
    • fine_amount is NULL until book is returned (calculated at return time)
    • returned_at is NULL for active loans
    CREATE DATABASE library_db;
USE library_db;

CREATE TABLE books (
  isbn            VARCHAR(20) PRIMARY KEY,
  title           VARCHAR(200) NOT NULL,
  author          VARCHAR(100) NOT NULL,
  genre           VARCHAR(50),
  total_copies    INT NOT NULL DEFAULT 1,
  available_copies INT NOT NULL DEFAULT 1,
  CHECK (available_copies >= 0),
  CHECK (available_copies <= total_copies)
);

CREATE TABLE members (
  id              INT PRIMARY KEY AUTO_INCREMENT,
  name            VARCHAR(100) NOT NULL,
  email           VARCHAR(150) UNIQUE,
  phone           VARCHAR(15),
  membership_type ENUM('student','faculty','public') DEFAULT 'student',
  expiry_date     DATE NOT NULL
);

CREATE TABLE loans (
  id          INT PRIMARY KEY AUTO_INCREMENT,
  book_isbn   VARCHAR(20) NOT NULL,
  member_id   INT NOT NULL,
  loaned_at   DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  due_date    DATE NOT NULL,
  returned_at DATETIME,
  fine_amount DECIMAL(8,2),
  FOREIGN KEY (book_isbn)  REFERENCES books(isbn),
  FOREIGN KEY (member_id)  REFERENCES members(id),
  CHECK (returned_at IS NULL OR returned_at >= loaned_at)
);

  3. Write analytical queries
    -- 1. Overdue loans (not returned past due date)
SELECT m.name, b.title, l.due_date,
  DATEDIFF(NOW(), l.due_date) AS days_overdue
FROM loans l
JOIN members m ON l.member_id = m.id
JOIN books   b ON l.book_isbn = b.isbn
WHERE l.returned_at IS NULL AND l.due_date < CURDATE()
ORDER BY days_overdue DESC;

-- 2. Most borrowed books this month
SELECT b.title, COUNT(l.id) AS loan_count
FROM loans l
JOIN books b ON l.book_isbn = b.isbn
WHERE l.loaned_at >= DATE_FORMAT(NOW(), '%Y-%m-01')
GROUP BY b.isbn, b.title
ORDER BY loan_count DESC;

-- 3. Members with outstanding fines (returned late)
SELECT m.name, SUM(l.fine_amount) AS total_fines
FROM loans l
JOIN members m ON l.member_id = m.id
WHERE l.fine_amount > 0
GROUP BY m.id, m.name
ORDER BY total_fines DESC;

-- 4. Books with 0 available copies (fully loaned out)
SELECT isbn, title, total_copies
FROM books
WHERE available_copies = 0;

-- 5. Member borrowing history
SELECT m.name, b.title, l.loaned_at, l.due_date,
  COALESCE(CAST(l.returned_at AS CHAR), 'Not returned') AS returned_at,
  COALESCE(l.fine_amount, 0) AS fine
FROM loans l
JOIN members m ON l.member_id = m.id
JOIN books   b ON l.book_isbn = b.isbn
WHERE m.id = 1
ORDER BY l.loaned_at DESC;

Complex SQL — Analytics Queries (Weeks 5–6)

▼

Objective

Write 10 analytical SQL queries on the e-commerce schema using window functions, CTEs, and advanced aggregations.

10 Analytical Queries

USE it_university;  -- switch back to e-commerce DB

-- 1. Monthly revenue trend for the past 12 months
SELECT
  DATE_FORMAT(created_at, '%Y-%m') AS month,
  COUNT(*)       AS orders,
  SUM(total)     AS revenue
FROM orders
WHERE status = 'delivered'
  AND created_at >= DATE_SUB(NOW(), INTERVAL 12 MONTH)
GROUP BY DATE_FORMAT(created_at, '%Y-%m')
ORDER BY month;

-- 2. Week-over-week order growth using LAG()
-- (Requires MySQL 8.0+ for window functions)
WITH weekly AS (
  SELECT
    YEARWEEK(created_at) AS week,
    COUNT(*) AS order_count
  FROM orders
  GROUP BY YEARWEEK(created_at)
)
SELECT week, order_count,
  LAG(order_count) OVER (ORDER BY week) AS prev_week,
  order_count - LAG(order_count) OVER (ORDER BY week) AS growth
FROM weekly;

-- 3. Top 5 products by revenue with running total
SELECT
  p.name,
  SUM(oi.quantity * oi.unit_price) AS product_revenue,
  SUM(SUM(oi.quantity * oi.unit_price))
    OVER (ORDER BY SUM(oi.quantity * oi.unit_price) DESC
          ROWS UNBOUNDED PRECEDING) AS running_total
FROM order_items oi
JOIN products p ON oi.product_id = p.id
GROUP BY p.id, p.name
ORDER BY product_revenue DESC
LIMIT 5;

-- 4. Customers ranked by lifetime value
SELECT
  c.name,
  SUM(o.total) AS lifetime_value,
  RANK() OVER (ORDER BY SUM(o.total) DESC) AS ranking
FROM customers c
JOIN orders o ON c.id = o.customer_id
GROUP BY c.id, c.name;

-- 5. Average order value by status
SELECT status, COUNT(*) AS orders, ROUND(AVG(total), 2) AS avg_value
FROM orders
GROUP BY status
ORDER BY avg_value DESC;

-- 6. Product categories contributing >10% of total revenue
WITH total AS (SELECT SUM(oi.quantity * oi.unit_price) AS grand_total FROM order_items oi),
     cat_rev AS (
       SELECT p.category,
         SUM(oi.quantity * oi.unit_price) AS cat_revenue
       FROM order_items oi
       JOIN products p ON oi.product_id = p.id
       GROUP BY p.category
     )
SELECT cat_rev.category, cat_rev.cat_revenue,
  ROUND(cat_rev.cat_revenue / total.grand_total * 100, 1) AS pct
FROM cat_rev, total
WHERE cat_rev.cat_revenue / total.grand_total > 0.10
ORDER BY cat_rev.cat_revenue DESC;

-- 7. Customers who bought in their first month but not their second (churn)
WITH first_order AS (
  SELECT customer_id, MIN(DATE_FORMAT(created_at,'%Y-%m')) AS first_month
  FROM orders GROUP BY customer_id
),
second_month_orders AS (
  SELECT DISTINCT o.customer_id
  FROM orders o
  JOIN first_order fo ON o.customer_id = fo.customer_id
  WHERE DATE_FORMAT(o.created_at,'%Y-%m') =
    DATE_FORMAT(DATE_ADD(STR_TO_DATE(CONCAT(fo.first_month,'-01'),'%Y-%m-%d'),
    INTERVAL 1 MONTH), '%Y-%m')
)
SELECT c.name, fo.first_month
FROM first_order fo
JOIN customers c ON fo.customer_id = c.id
WHERE fo.customer_id NOT IN (SELECT customer_id FROM second_month_orders);

-- 8. Median order value (MySQL has no MEDIAN())
SET @row_index := -1;
SELECT AVG(subq.total) AS median_order_value
FROM (
  SELECT @row_index := @row_index + 1 AS row_index, total
  FROM orders ORDER BY total
) AS subq
WHERE subq.row_index IN (FLOOR(@row_index / 2), CEIL(@row_index / 2));

-- 9. Recursive CTE: category hierarchy (self-referential)
CREATE TABLE IF NOT EXISTS categories (
  id        INT PRIMARY KEY AUTO_INCREMENT,
  name      VARCHAR(100),
  parent_id INT,
  FOREIGN KEY (parent_id) REFERENCES categories(id)
);
INSERT INTO categories (id, name, parent_id) VALUES
  (1, 'Electronics', NULL), (2, 'Computers', 1),
  (3, 'Laptops', 2), (4, 'Gaming Laptops', 3), (5, 'Mobile', 1);

WITH RECURSIVE cat_tree AS (
  SELECT id, name, parent_id, 0 AS depth, CAST(name AS CHAR(500)) AS path
  FROM categories WHERE parent_id IS NULL
  UNION ALL
  SELECT c.id, c.name, c.parent_id, ct.depth+1,
    CONCAT(ct.path, ' > ', c.name)
  FROM categories c JOIN cat_tree ct ON c.parent_id = ct.id
)
SELECT depth, name, path FROM cat_tree ORDER BY path;

-- 10. Order items with % contribution to order total
SELECT
  o.id AS order_id,
  p.name AS product,
  oi.quantity * oi.unit_price AS item_total,
  o.total AS order_total,
  ROUND((oi.quantity * oi.unit_price) / o.total * 100, 1) AS pct_of_order
FROM order_items oi
JOIN orders   o ON oi.order_id  = o.id
JOIN products p ON oi.product_id = p.id
ORDER BY o.id, item_total DESC;

Stored Procedures & Triggers (Week 8)

▼

Objective

Build 3 stored procedures and 2 triggers, then verify their behavior through testing.

Setup: Audit Log Table

CREATE TABLE IF NOT EXISTS audit_log (
  id         INT PRIMARY KEY AUTO_INCREMENT,
  action     VARCHAR(100) NOT NULL,
  table_name VARCHAR(50),
  record_id  INT,
  user_id    INT,
  details    TEXT,
  created_at DATETIME DEFAULT CURRENT_TIMESTAMP
);

Stored Procedure 1: sp_place_order

DELIMITER //
CREATE PROCEDURE sp_place_order(
  IN  p_customer_id INT,
  IN  p_cart        JSON,      -- [{"product_id":1,"qty":2},{"product_id":3,"qty":1}]
  OUT p_order_id    INT,
  OUT p_message     VARCHAR(200)
)
BEGIN
  DECLARE v_total     DECIMAL(10,2) DEFAULT 0;
  DECLARE v_price     DECIMAL(10,2);
  DECLARE v_stock     INT;
  DECLARE v_product   INT;
  DECLARE v_qty       INT;
  DECLARE v_idx       INT DEFAULT 0;
  DECLARE v_count     INT;

  DECLARE EXIT HANDLER FOR SQLEXCEPTION
  BEGIN
    ROLLBACK;
    SET p_order_id = -1;
    SET p_message  = 'Order failed — rolled back.';
  END;

  SET v_count = JSON_LENGTH(p_cart);
  START TRANSACTION;

  -- Validate stock and calculate total
  WHILE v_idx < v_count DO
    SET v_product = JSON_UNQUOTE(JSON_EXTRACT(p_cart, CONCAT('$[',v_idx,'].product_id')));
    SET v_qty     = JSON_UNQUOTE(JSON_EXTRACT(p_cart, CONCAT('$[',v_idx,'].qty')));

    SELECT price, stock INTO v_price, v_stock
    FROM products WHERE id = v_product FOR UPDATE;

    IF v_stock < v_qty THEN
      ROLLBACK;
      SET p_order_id = -1;
      SET p_message  = CONCAT('Insufficient stock for product ID ', v_product);
      LEAVE sp;
    END IF;

    UPDATE products SET stock = stock - v_qty WHERE id = v_product;
    SET v_total = v_total + (v_price * v_qty);
    SET v_idx = v_idx + 1;
  END WHILE;

  -- Create order
  INSERT INTO orders (customer_id, total, status) VALUES (p_customer_id, v_total, 'pending');
  SET p_order_id = LAST_INSERT_ID();

  -- Insert order items
  SET v_idx = 0;
  WHILE v_idx < v_count DO
    SET v_product = JSON_UNQUOTE(JSON_EXTRACT(p_cart, CONCAT('$[',v_idx,'].product_id')));
    SET v_qty     = JSON_UNQUOTE(JSON_EXTRACT(p_cart, CONCAT('$[',v_idx,'].qty')));
    SELECT price INTO v_price FROM products WHERE id = v_product;
    INSERT INTO order_items (order_id, product_id, quantity, unit_price)
    VALUES (p_order_id, v_product, v_qty, v_price);
    SET v_idx = v_idx + 1;
  END WHILE;

  COMMIT;
  SET p_message = CONCAT('Order ', p_order_id, ' placed. Total: ₹', v_total);
END //
DELIMITER ;

Stored Procedure 2: sp_get_customer_summary

DELIMITER //
CREATE PROCEDURE sp_get_customer_summary(IN p_customer_id INT)
BEGIN
  SELECT
    c.name,
    c.email,
    c.city,
    COUNT(o.id)   AS total_orders,
    SUM(o.total)  AS lifetime_value,
    MAX(o.created_at) AS last_order_date
  FROM customers c
  LEFT JOIN orders o ON c.id = o.customer_id
  WHERE c.id = p_customer_id
  GROUP BY c.id, c.name, c.email, c.city;
END //
DELIMITER ;

CALL sp_get_customer_summary(1);

Trigger 1: Audit Log on Order Insert

DELIMITER //
CREATE TRIGGER after_order_insert
AFTER INSERT ON orders
FOR EACH ROW
BEGIN
  INSERT INTO audit_log (action, table_name, record_id, details)
  VALUES (
    'ORDER_PLACED',
    'orders',
    NEW.id,
    CONCAT('Customer ', NEW.customer_id, ' placed order for ₹', NEW.total)
  );
END //
DELIMITER ;

Trigger 2: Prevent Deleting Products with Orders

DELIMITER //
CREATE TRIGGER before_product_delete
BEFORE DELETE ON products
FOR EACH ROW
BEGIN
  DECLARE v_count INT;
  SELECT COUNT(*) INTO v_count FROM order_items WHERE product_id = OLD.id;
  IF v_count > 0 THEN
    SIGNAL SQLSTATE '45000'
    SET MESSAGE_TEXT = 'Cannot delete a product that has existing order history';
  END IF;
END //
DELIMITER ;

EXPLAIN & Query Optimisation (Week 6)

▼

Objective

Take 5 deliberately slow queries, diagnose them with EXPLAIN, apply fixes, and measure the improvement.

5 Slow Queries and Their Fixes

-- =============================================
-- Problem 1: Function on indexed column
-- =============================================
-- ✗ SLOW: YEAR() prevents index use
EXPLAIN SELECT * FROM orders WHERE YEAR(created_at) = 2024;
-- type=ALL, key=NULL

-- ✓ FIX: range condition uses index
CREATE INDEX idx_created ON orders(created_at);
EXPLAIN SELECT * FROM orders
WHERE created_at BETWEEN '2024-01-01 00:00:00' AND '2024-12-31 23:59:59';
-- type=range, key=idx_created

-- =============================================
-- Problem 2: Leading wildcard LIKE
-- =============================================
-- ✗ SLOW: leading % cannot use B-tree index
EXPLAIN SELECT * FROM customers WHERE email LIKE '%@gmail.com';
-- type=ALL

-- ✓ FIX: trailing wildcard can use index
CREATE INDEX idx_email ON customers(email);
EXPLAIN SELECT * FROM customers WHERE email LIKE 'priya%';
-- type=range, uses idx_email

-- For suffix search, use FULLTEXT index instead:
ALTER TABLE customers ADD FULLTEXT INDEX ft_email (email);

-- =============================================
-- Problem 3: Multi-table JOIN with no FK indexes
-- =============================================
EXPLAIN
SELECT c.name, p.name, oi.quantity
FROM order_items oi
JOIN orders   o ON oi.order_id   = o.id
JOIN customers c ON o.customer_id = c.id
JOIN products  p ON oi.product_id = p.id;
-- Without indexes on FK columns: all=ALL on every table

-- ✓ FIX: add indexes on all FK columns
CREATE INDEX idx_oi_order   ON order_items(order_id);
CREATE INDEX idx_oi_product ON order_items(product_id);
CREATE INDEX idx_ord_cust   ON orders(customer_id);

EXPLAIN
SELECT c.name, p.name, oi.quantity
FROM order_items oi
JOIN orders   o ON oi.order_id   = o.id
JOIN customers c ON o.customer_id = c.id
JOIN products  p ON oi.product_id = p.id;
-- Now: ref/eq_ref for all joins

-- =============================================
-- Problem 4: SELECT * with unnecessary columns
-- =============================================
-- ✗ SLOW: fetches all columns, prevents covering index
EXPLAIN SELECT * FROM customers WHERE city = 'Delhi';

-- ✓ FIX: select only needed columns, create covering index
CREATE INDEX idx_city_name ON customers(city, name, email);
EXPLAIN SELECT name, email FROM customers WHERE city = 'Delhi';
-- Extra column: Using index (covering index — no table lookup!)

-- =============================================
-- Problem 5: ORDER BY on non-indexed column causing filesort
-- =============================================
EXPLAIN SELECT * FROM orders ORDER BY total DESC LIMIT 10;
-- Extra: Using filesort (sorts entire result in memory/disk — slow)

CREATE INDEX idx_total ON orders(total);
EXPLAIN SELECT id, customer_id, total FROM orders ORDER BY total DESC LIMIT 10;
-- Extra: Using index (reads index in order — no sort needed)

  4. JSON queries
    -- Filter by JSONB category
SELECT title, meta->>'category' AS category
FROM posts
WHERE meta->>'category' = 'database';

-- Filter by JSONB featured flag
SELECT title FROM posts WHERE meta @> '{"featured": true}';

-- Sort by views (cast text to int)
SELECT title, (meta->>'views')::int AS views
FROM posts
ORDER BY views DESC;

-- GIN index for faster JSONB queries
CREATE INDEX idx_meta_gin ON posts USING GIN(meta);

-- Verify index is used
EXPLAIN SELECT * FROM posts WHERE meta @> '{"category": "database"}';
  5. Array queries
    -- Posts with 'nodejs' tag
SELECT title FROM posts WHERE 'nodejs' = ANY(tags);

-- Posts with ALL of these tags
SELECT title FROM posts WHERE tags @> ARRAY['database','postgresql'];

-- Expand tags into individual rows
SELECT id, title, unnest(tags) AS tag FROM posts ORDER BY id, tag;
  6. Window functions
    -- Rank posts by views per category
SELECT
  title,
  meta->>'category' AS category,
  (meta->>'views')::int AS views,
  RANK() OVER (
    PARTITION BY meta->>'category'
    ORDER BY (meta->>'views')::int DESC
  ) AS rank_in_category
FROM posts;

-- Previous post's view count with LAG()
SELECT
  title,
  created_at,
  (meta->>'views')::int AS views,
  LAG((meta->>'views')::int) OVER (ORDER BY created_at) AS prev_views
FROM posts
ORDER BY created_at;

-- Cumulative views over time
SELECT
  title,
  created_at,
  (meta->>'views')::int AS views,
  SUM((meta->>'views')::int) OVER (ORDER BY created_at) AS cumulative_views
FROM posts
ORDER BY created_at;
